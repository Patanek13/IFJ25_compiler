import "ifj25" for Ifj
class Program {

static main() {
    var n
    var r

    n = 8
    r = fib(n)

    __d = Ifj.write("fib(")
    __d = Ifj.write(n)
    __d = Ifj.write(") = ")
    __d = Ifj.write(r)
    __d = Ifj.write("\n")
}

static fib(n) {
    var a
    var b
    var i
    var tmp

    if (n <= 0) {
        return 0
    } else {
        a = 0
        b = 1
        i = 1
        while (i < n) {
            tmp = a + b
            a = b
            b = tmp
            i = i + 1
        }
        return b
    }
}
}
