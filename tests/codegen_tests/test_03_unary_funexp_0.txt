import "ifj25" for Ifj
class Program {

    static getBool() {
        return false
    }
    static getNum() {
        return 5
    }
    static main() {
        var b
        b = !getBool()
        Ifj.write(b)
        Ifj.write("\n")
        var i
        i = -getNum()
        Ifj.write(i)
        Ifj.write("\n")
    }

}
