import "ifj25" for Ifj
class Program {

static main() {
    var s
    var r
    var len
    var i
    var code
    var ch

    s = "abcd"
    r = ""
    len = Ifj.length(s)

    // index posledného znaku
    i = len - 1

    while (i >= 0) {
        code = Ifj.ord(s, i)
        ch = Ifj.chr(code)
        r = r + ch
        i = i - 1
    }

    __d = Ifj.write("orig=")
    __d = Ifj.write(s)
    __d = Ifj.write("\n")

    __d = Ifj.write("rev=")
    __d = Ifj.write(r)
    __d = Ifj.write("\n")
}
}
