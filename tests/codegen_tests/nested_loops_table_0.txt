import "ifj25" for Ifj
class Program {

static main() {
    var i
    var j
    var prod

    i = 1
    while (i <= 3) {
        j = 1
        while (j <= 3) {
            prod = i * j

            __d = Ifj.write(i)
            __d = Ifj.write(" x ")
            __d = Ifj.write(j)
            __d = Ifj.write(" = ")
            __d = Ifj.write(prod)
            __d = Ifj.write("\n")

            j = j + 1
        }
        i = i + 1
    }
}
}
